//POST API Create Account
app.post("/createAccount", function(req, res){
let accountDetails = {};
accountDetails.email = req.body.email;
accountDetails.email = req.body.password;
//{email = req.body.email, password = req.body.password};
accountDetails.id = database.exeQuery("exec dbo.insertuser '" + accountDetails.email + "','" + accountDetails.password +"'");
return res.send(accountDetails);
});
